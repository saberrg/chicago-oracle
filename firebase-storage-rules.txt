rules_version = '2';

service firebase.storage {
  match /b/{bucket}/o {
    // Allow anyone to read images, but only authenticated users can write
    match /images/{allPaths=**} {
      allow read: if true;
      allow write: if request.auth != null;
    }
    
    // Deny all other paths
    match /{allPaths=**} {
      allow read, write: if false;
    }
  }
}
